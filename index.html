uiCalendar / minimalistic JS + HTML5 + CSS3 datepicker

Container

uiCalendar({
	container: document.getElementById('dev-wrapper'),
	inputFrom: document.getElementById('input-container-from'),
	inputTo: document.getElementById('input-container-to'),
	range: true,
	monthCount: 3,
	fillSpaces: false
});
